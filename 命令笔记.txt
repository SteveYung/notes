1.git删除本地分支
git branch -d googleplay_ledou
git branch -D googleplay_ledou

2.git删除远程分支
git push origin :taptap_ledou

3.git拉取远程分支
git checkout origin/googleplay_ledou 
git checkout -b googleplay_ledou 

4.修改git提交名和邮箱
git config --global user.name "your name"
git config --global user.email "your email"

5.git设置代理
git config --global http.proxy 'socks5://127.0.0.1:1080'
git config --global https.proxy 'socks5://127.0.0.1:1080'
